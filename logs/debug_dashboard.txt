    <script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>
    <script>
        let map;
        let detectionsLayer;
        let zonesLayer;

        function initMap() {
            // Vue initiale centrée sur Treichville
            map = L.map('map', {
                zoomControl: false // Disable default zoom to match UI
            }).setView([5.30, -4.00], 14);

            // Google Maps Satellite Layout
            L.tileLayer('https://mt1.google.com/vt/lyrs=s&x={x}&y={y}&z={z}', {
                attribution: 'Tiles &copy; Google',
                maxZoom: 20
            }).addTo(map);

            loadZonesCadastrales();
            loadDetections();
        }

        function loadZonesCadastrales() {
            fetch('/api/zones-geojson/')
                .then(r => r.json())
                .then(data => {
                    zonesLayer = L.geoJSON(data, {
                        style: function (feature) {
                            return {
                                fillColor: '#0088ff',
                                weight: 1,
                                opacity: 0.5,
                                color: '#0088ff',
                                fillOpacity: 0.1
                            };
                        }
                    }).addTo(map);
                });
        }

        function loadDetections() {
            fetch('/api/detections-geojson/')
                .then(r => r.json())
                .then(data => {
                    const alertColors = {
                        'rouge': '#FF4444',
                        'orange': '#fd7e14',
                        'vert': '#00FF88',
                        'veille': '#0dcaf0'
                    };

                    detectionsLayer = L.geoJSON(data, {
                        style: function (feature) {
                            const color = alertColors[feature.properties.alert_level] || '#888888';
                            return {
                                fillColor: color,
                                color: color,
                                weight: 2,
                                opacity: 1,
                                fillOpacity: 0.4
                            };
                        },
                        onEachFeature: function (feature, layer) {
                            const p = feature.properties;
                            const lat = p.latitude != null ? p.latitude.toFixed(6) : 'N/A';
                            const lon = p.longitude != null ? p.longitude.toFixed(6) : 'N/A';

                            const gmapLink = (p.latitude && p.longitude) ?
                                `<a href="https://www.google.com/maps/search/?api=1&query=${lat},${lon}" target="_blank" class="cyber-link mb-2" style="display:block; text-align:center;"><i class="fas fa-satellite"></i> GOOGLE MAPS</a>` : '';

                            layer.bindPopup(`
                                <div style="font-size:11px; padding:5px;">
                                    <strong style="color:var(--neon-green)">[ DÉTECTION #${p.id} ]</strong><br><br>
                                    STATUT: ${p.alert_label || p.status}<br>
                                    ZONE: ${p.zone_name || 'HORS CADASTRE'}<br>
                                    GPS: ${lat}, ${lon}<br><br>
                                    ${gmapLink}
                                    <button onclick="navigator.clipboard.writeText('${lat}, ${lon}').then(()=>alert('Coordonnées copiées !'))" class="cyber-btn" style="width:100%; height:30px; margin-bottom:10px; font-size:10px; color:var(--neon-green)">
                                        <i class="fas fa-copy"></i> COPIER GPS
                                    </button>
                                    <a href="/detections/${p.id}/" class="cyber-link" style="width:100%; text-align:center; box-sizing:border-box; display:block;">>> INVESTIGUER</a>
                                </div>
                            `);
                        }
                    }).addTo(map);
                });
        }

        function setActive(element) {
            document.querySelectorAll('.sidebar-left .cyber-btn').forEach(btn => btn.classList.remove('active'));
            element.classList.add('active');
        }

        function showAllDetections() {
            if (detectionsLayer) {
                detectionsLayer.eachLayer(layer => {
                    layer.setStyle({ opacity: 1, fillOpacity: 0.4 });
                });
            }
        }

        function showRedAlerts() {
            if (detectionsLayer) {
                detectionsLayer.eachLayer(layer => {
                    let isRed = layer.feature.properties.alert_level === 'rouge';
                    layer.setStyle({
                        opacity: isRed ? 1 : 0.05,
                        fillOpacity: isRed ? 0.6 : 0.0
                    });
                });
            }
        }

        function showOrangeAlerts() {
            if (detectionsLayer) {
                detectionsLayer.eachLayer(layer => {
                    let isOrange = layer.feature.properties.alert_level === 'orange';
                    layer.setStyle({
                        opacity: isOrange ? 1 : 0.05,
                        fillOpacity: isOrange ? 0.6 : 0.0
                    });
                });
            }
        }

        document.addEventListener('DOMContentLoaded', initMap);
    </script>
